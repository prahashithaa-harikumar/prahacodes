@RestController
@RequestMapping("/api/project/credit-applications")
public class CreditApplicationController {

    @Autowired
    private CreditApplicationService creditApplicationService;

    @PostMapping("/process")
    public ResponseEntity<String> processCustomerApplication(@RequestBody CustomerDataDto customerDataDto) {
        try {
            creditApplicationService.processCustomerData(customerDataDto);
            return ResponseEntity.ok("Application processed successfully");
        } catch (Exception e) {
            return ResponseEntity.status(HttpStatus.INTERNAL_SERVER_ERROR).body("Error processing application: " + e.getMessage());
        }
    }
}
