@GetMapping("/customers/{customerId}")
public ResponseEntity<?> getCustomerById(@PathVariable Long customerId) {
    try {
        // Try to fetch the customer by ID
        Customer customer = customerService.getCustomerById(customerId);
        
        if (customer != null) {
            // Return the customer details if found
            return ResponseEntity.ok(customer);
        } else {
            // If customer is not found, return a 404 Not Found status
            return ResponseEntity.status(HttpStatus.NOT_FOUND).body("Customer not found with ID: " + customerId);
        }
    } catch (Exception e) {
        // Catch any exceptions and return a 500 Internal Server Error
        return ResponseEntity.status(HttpStatus.INTERNAL_SERVER_ERROR).body("An error occurred: " + e.getMessage());
    }
}
