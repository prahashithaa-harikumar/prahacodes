import org.springframework.data.jpa.repository.JpaRepository;
import org.springframework.data.jpa.repository.Query;
import org.springframework.data.repository.query.Param;

import java.util.Optional;

public interface CustomerRepository extends JpaRepository<Customer, Long> {

    @Query("SELECT new com.example.demo.CustomerDTO(c.customerId, c.name, c.dateOfBirth, c.address, c.email, c.contact, " +
            "c.adhaarNumber, c.panNumber, c.monthlyIncome) " +
            "FROM Customer c WHERE c.customerId = :customerId")
    Optional<CustomerDTO> findCustomerDetailsById(@Param("customerId") Long customerId);
}
